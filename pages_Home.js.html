import React, { useState, useEffect } from 'react';
import axios from 'axios';
import logo from './../icecream.gif'
import './Home.css'


const pageStyle = {
  backgroundColor: '#f7efde',
  minHeight: '100vh',
};

const containerStyle = {
  backgroundColor: '#f7efde',
  padding: '20px', 
  textAlign: 'center',
};

const imageContainerStyle = {
  display: 'flex',
  justifyContent: 'center',
};

const imageStyle = {
  width: '200px', 
  margin: '10px', 
};

const textContainerStyle = {
  marginTop: '20px',
  marginBottom: '50px'
};

/**
 * Home Component
 * 
 * This component serves as the homepage for the Lick Honest Ice Creams website. It displays
 * a welcome message, an image, information about the ice cream shop, and the current weather
 * at the shop's location.
 * 
 * State Variables:
 * - weather: The current weather data fetched from an API.
 * 
 * @returns {React.Component} The Home component.
 */
function Home() {
  const [weather, setWeather] = useState(null);

  /**
   * Fetches the current weather data from the Open-Meteo API.
   * Uses the coordinates for College Station, TX.
   */
  useEffect(() => {
    const fetchWeather = async () => {
      try {
        // Using College Station, TX coordinates: Latitude 30.6280, Longitude -96.3344
        const response = await axios.get('https://api.open-meteo.com/v1/forecast', {
          params: {
            latitude: 30.6280,
            longitude: -96.3344,
            current_weather: true
          }
        });
        setWeather(response.data.current_weather);
      } catch (error) {
        console.error("Error fetching weather data:", error);
      }
    };

    fetchWeather();
  }, []);

  /**
   * Converts a temperature from Celsius to Fahrenheit.
   * 
   * @param {number} celsius - The temperature in Celsius.
   * @returns {number} The temperature in Fahrenheit.
   */
  const convertCelsiusToFahrenheit = (celsius) => {
    return (celsius * 9/5) + 32;
  };

  return (
    <div style={pageStyle}>
    <div style={containerStyle}>
      <h1>Welcome to Lick Honest Ice Creams!</h1>
      <div style={imageContainerStyle}>
        <img src={logo} alt="dancing ice cream man" style={imageStyle} />
      </div>
      <p className="about-lick">
        To us, "honest" simply means we use the purest ingredients we can find in our artisanal ice creams. 
        What doesn’t go into our ice creams is just as important as what does. We never use artificial colors or 
        flavors, high fructose corn syrup or preservatives, ever! We can trace every single ingredient we use to 
        its source. Ice cream shouldn’t just taste good — we believe it should also be good.
      </p>
      <div style={textContainerStyle}>
        <p> <strong>Address</strong><br /> 
            143 Century Square Dr, Suite 11,<br /> 
            College Station, TX 77840</p>
        <p> <strong>Hours</strong><br />
            Sun - Thu: 12:30pm 'til 10pm <br /> 
            Fri - Sat: 12pm 'til 11pm</p>
        <p> <strong>Phone Number</strong><br />
            (979) 704-0306</p>
      </div>

      <div className="weather">
          {weather && (
            <div>
              <h2>Current Weather</h2>
              <p>Temperature: {convertCelsiusToFahrenheit(weather.temperature).toFixed(1)} °F</p>
              <p>Wind Speed: {weather.windspeed} km/h</p>
            </div>
          )}
        </div>
    </div>
    </div>
    
  );
}

export default Home;
